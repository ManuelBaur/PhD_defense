\begin{textblock}{0.9}(0.05,0.1)
	\only<1>{
	\begin{align}
	D(q,\tau)
	&= \Big\langle |I(q, t+\tau) - I(q,t)|^2  \Big\rangle_t \nonumber \\
	&\textcolor{white}{
	= A(q) 
	\bigg[1-
	\frac{\big\langle I^*(q,t) I(q,t+\tau) \big\rangle_t}
	{\big\langle |I(q,t)|^2 \big\rangle_t}
	\bigg] 
	+ B(q)}
	\nonumber
	\end{align}
	}
	\only<2>{
	\begin{align}
	D(q,\tau)
	&= \Big\langle |I(q, t+\tau) - I(q,t)|^2  \Big\rangle_t \nonumber \\
	&= A(q) 
	\bigg[1-
			\frac{\big\langle I^*(q,t) I(q,t+\tau) \big\rangle_t}
			{\big\langle |I(q,t)|^2 \big\rangle_t}
	\bigg] 
	+ B(q)\nonumber
	\end{align}
	}
	\visible<3->{
	\begin{align}
	D(q,\tau)
	&= \Big\langle |I(q, t+\tau) - I(q,t)|^2  \Big\rangle_t \nonumber \\
	&= A(q) 
	\bigg[1-
	\textcolor{red}{\underbrace{
			\frac{\big\langle I^*(q,t) I(q,t+\tau) \big\rangle_t}
			{\big\langle |I(q,t)|^2 \big\rangle_t}
	}}
	\bigg] 
	+ B(q)\nonumber
	\end{align}
	}
